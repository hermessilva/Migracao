  // Enum que define os possíveis status de um seguro durante seu ciclo de vida
  enum status_seguro {
    "Em análise pela Seguradora" // Proposta enviada e aguardando aprovação da seguradora
    "Pendente de documentação" // Aguardando envio ou validação de documentos obrigatórios
    "Ativo" // Seguro vigente e em operação normal
    "Expiração da vigência do seguro" // Seguro encerrado por término do prazo de vigência
    "Cancelado pelo cooperado" // Cancelamento solicitado pelo próprio cooperado
    "Cancelado pela cooperativa" // Cancelamento realizado pela cooperativa
    "Sinistro" // Seguro encerrado por ocorrência de sinistro (morte, invalidez, etc.)
    "Recusado pela seguradora" // Proposta negada pela seguradora
    "Cancelamento por prejuízo" // Cancelamento devido a inadimplência ou prejuízo financeiro
    "Liquidação antecipada" // Contrato quitado antes do prazo previsto
    "Cancelado por renegociação" // Cancelamento devido à renegociação do contrato vinculado
    "Cancelado por aditivo" // Cancelamento devido a aditivo contratual
}

// Enum que define as modalidades de pagamento do seguro
enum tipo_pagamento {
     "À Vista" // Pagamento integral no ato da contratação
     "Parcelado" // Pagamento dividido em parcelas mensais
     "Único" // Pagamento único incluído no contrato de crédito
}
